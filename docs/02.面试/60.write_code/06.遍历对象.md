# 遍历对象

```js
  var obj = {
          a: {
              b: 1,
              c: 2,
              d: {
                  e: 5
              }
          },
          b: [1, 3, {
              a: 2,
              b: 3
          }],
          c: 3
      }
      
  // flatten(obj) 结果返回如下
  //{'a.b':1,'a.c':2,'a.d.e':5,'b[0]':1,'b[1]':3,'b[2].a':2,'b[2].b':3,c:3}

  function isObject(val) {
      return typeof val === 'object' && val !== null
  }

  function flatten() {
      let res = {}
      const dfs = (cur, prefix) => {
          for (let k in cur) {
              // 如果是数组或是对象
              if (isObject(cur[k])) {
                  if (Array.isArray(cur[k])) {
                      cur[k].forEach((item, index) => {
                          if (isObject(item)) {
                              dfs(item, prefix + k + "[" + index + "]" + ".")
                          } else {
                              res[prefix + k + "[" + index + "]"] = item
                          }
                      });
                  } else {
                      dfs(cur[k], prefix + k + ".")
                  }
              } else {
                  res[prefix + k] = cur[k]
              }
          }
      }
      dfs(obj, "")
      return res
  }
  console.log(flatten())
```
