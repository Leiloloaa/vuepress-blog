# 去重数组对象并整合值为数组

> 数组对象中特定值相同的合并，并且将其同级属性的值合并成数组

```javascript
  (function resetList() {
            let list = [{
                ae: '1',
                tid: 2
            }, {
                ae: '2',
                tid: 8
            }, {
                ae: '1',
                tid: 4
            }, {
                ae: '1',
                tid: 5
            }, {
                ae: '2',
                tid: 5
            }, {
                ae: '3',
                tid: 5
            }]
            let result = [],
                obj = {};
            let i = 0;
            for (i in list) {
                if (!obj[list[i].ae]) {
                    result.push(list[i])
                    obj[list[i].ae] = 1
                } else {
                    obj[list[i].ae]++
                }
            }
            let j = 0
            let k = 0
            for (k in result) {
                let flag = true
                for (j in list) {
                    if (list[j].ae == result[k].ae && list[j].tid != result[k].tid) {
                        let temp = result[k].tid
                        result[k].tid = Array.from(result[k].tid)
                        if (flag) {
                            result[k].tid.push(temp)
                        }
                        result[k].tid.push(list[j].tid)
                        flag = false
                    }
                }
            }
            console.log(result)
        })();
```
