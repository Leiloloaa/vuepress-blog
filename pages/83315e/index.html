    database: "shopDB"

    console.log('3000...')

// 商品操作---------------------------------
// 新增商品
app.get("/api/createOrUpdateGoods", async(req, res) => {
    let pBody = { id: 3, name: "iphone 13" };
    let sql = "INSERT INTO goods SET ?";
    db.query(sql, pBody, (err, result) => {
        if (err) {
            console.log(err);
        } else {
            res.send('插入成功')
            console.log(result);
        }
    })
})

// 商品列表
app.get("/api/getGoodsList", async(req, res) => {
    let sql = "select * from goods";

        if (err) {
            console.log(err);
        } else {
            console.log(result);
            res.send(result);
        }
    })
});
